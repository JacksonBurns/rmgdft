53 static inline double
54 cheb_eval(const double x, const double *cs, const int N)
55 {
56  int i;
57  double twox, b0, b1, b2;
58 
59  b2 = b1 = b0 = 0.0;
60 
61  twox = 2.0*x;
62  for(i=N-1; i>=0; i--){
63  b2 = b1;
64  b1 = b0;
65  b0 = twox*b1 - b2 + cs[i];
66  }
67 
68  return 0.5*(b0 - b2);
69 }
